443  template <typename U>
444  void draw_circle(const basic_point<U>& position, const float radius) noexcept
445  {
446  using value_t = typename basic_point<U>::value_type;
447 
448  auto error = -radius;
449  auto x = radius - 0.5f;
450  auto y = 0.5f;
451 
452  const auto cx = static_cast<float>(position.x()) - 0.5f;
453  const auto cy = static_cast<float>(position.y()) - 0.5f;
454 
455  while (x >= y)
456  {
457  draw_point<value_t>({static_cast<value_t>(cx + x), static_cast<value_t>(cy + y)});
458  draw_point<value_t>({static_cast<value_t>(cx + y), static_cast<value_t>(cy + x)});
459 
460  if (x != 0)
461  {
462  draw_point<value_t>({static_cast<value_t>(cx - x), static_cast<value_t>(cy + y)});
463  draw_point<value_t>({static_cast<value_t>(cx + y), static_cast<value_t>(cy - x)});
464  }
465 
466  if (y != 0)
467  {
468  draw_point<value_t>({static_cast<value_t>(cx + x), static_cast<value_t>(cy - y)});
469  draw_point<value_t>({static_cast<value_t>(cx - y), static_cast<value_t>(cy + x)});
470  }
471 
472  if (x != 0 && y != 0)
473  {
474  draw_point<value_t>({static_cast<value_t>(cx - x), static_cast<value_t>(cy - y)});
475  draw_point<value_t>({static_cast<value_t>(cx - y), static_cast<value_t>(cy - x)});
476  }
477 
478  error += y;
479  ++y;
480  error += y;
481 
482  if (error >= 0)
483  {
484  --x;
485  error -= x;
486  error -= x;
487  }
488  }
489  }
