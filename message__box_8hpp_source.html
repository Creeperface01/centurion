488  auto show(SDL_Window* parent) -> std::optional<button_id>
489  {
490  SDL_MessageBoxData data{};
491 
492  data.window = parent;
493  data.title = m_title.c_str();
494  data.message = m_message.c_str();
495  data.flags = to_flags(m_type, m_buttonOrder);
496  data.colorScheme = m_colorScheme ? m_colorScheme->get() : nullptr;
497 
498 #if CENTURION_HAS_FEATURE_MEMORY_RESOURCE
499  // Realistically 1-3 buttons, stack buffer for 8 buttons, just in case.
500  detail::stack_resource<8 * sizeof(SDL_MessageBoxButtonData)> resource;
501  std::pmr::vector<SDL_MessageBoxButtonData> buttonData{resource.get()};
502 #else
503  std::vector<SDL_MessageBoxButtonData> buttonData;
504  buttonData.reserve(8);
505 #endif // CENTURION_HAS_STD_MEMORY_RESOURCE
506 
507  if (m_buttons.empty())
508  {
509  add_button(0, "OK", default_button::return_key);
510  }
511 
512  for (const auto& button : m_buttons)
513  {
514  buttonData.emplace_back(button.convert());
515  }
516 
517  data.buttons = buttonData.data();
518  data.numbuttons = isize(buttonData);
519 
520  button_id button{-1};
521  if (SDL_ShowMessageBox(&data, &button) == -1)
522  {
523  throw sdl_error{};
524  }
525 
526  if (button != -1)
527  {
528  return button;
529  }
530  else
531  {
532  return std::nullopt;
533  }
534  }
