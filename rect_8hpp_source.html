751 template <typename T>
752 [[nodiscard]] constexpr auto get_union(const basic_rect<T>& fst,
753  const basic_rect<T>& snd) noexcept -> basic_rect<T>
754 {
755  const auto fstHasArea = fst.has_area();
756  const auto sndHasArea = snd.has_area();
757 
758  if (!fstHasArea && !sndHasArea)
759  {
760  return {};
761  }
762  else if (!fstHasArea)
763  {
764  return snd;
765  }
766  else if (!sndHasArea)
767  {
768  return fst;
769  }
770 
771  const auto x = detail::min(fst.x(), snd.x());
772  const auto y = detail::min(fst.y(), snd.y());
773  const auto maxX = detail::max(fst.max_x(), snd.max_x());
774  const auto maxY = detail::max(fst.max_y(), snd.max_y());
775 
776  return {{x, y}, {maxX - x, maxY - y}};
777 }
