728  const basic_rect<T>& snd) noexcept
729  -> basic_rect<T>
730 {

733 
734  if (!fstHasArea && !sndHasArea)
735  {
736  return {};

738  else if (!fstHasArea)

741  }
742  else if (!sndHasArea)

746

749  const auto maxX = detail::max(fst.max_x(), snd.max_x());
