358 [[nodiscard]] auto distance(const basic_point<T> from, const basic_point<T> to) noexcept ->
359  typename point_traits<T>::value_type
360 {
361  if constexpr (basic_point<T>::isIntegral) {
362  const auto xDiff = std::abs(from.x() - to.x());
363  const auto yDiff = std::abs(from.y() - to.y());
364  const auto dist = std::sqrt(xDiff + yDiff);
365  return static_cast<int>(std::round(dist));
366  }
367  else {
368  return std::sqrt(std::abs(from.x() - to.x()) + std::abs(from.y() - to.y()));
369  }
