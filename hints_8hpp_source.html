225 template <class Derived>
226 class enum_hint
227 {
228  public:
229  using value = typename enum_hint_traits<Derived>::value;
230 
231  template <typename T>
232  constexpr static auto valid_arg() noexcept -> bool
233  {
234  return std::is_same_v<T, value>;
235  }
236 
237  static auto current_value() noexcept -> std::optional<value>
238  {
239  czstring hint = SDL_GetHint(Derived::name());
240  if (!hint)
241  {
242  return std::nullopt;
243  }
244  return Derived::map.key_from(hint);
245  }
246 
247  static auto to_string(const value value) -> std::string
248  {
249  return Derived::map.find(value);
250  }
251 };
